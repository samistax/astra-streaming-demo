CREATE TABLE IF NOT EXISTS price_update (
    id text,
    itemName text,
    affiliation text,
    coupon text,
    discount double,
    idx int,
    itemBrand text,
    itemCategory text,
    itemCategory2 text,
    itemCategory3 text,
    itemCategory4 text,
    itemCategory5 text,
    itemListId text,
    itemListName text,
    itemVariant text,
    locationId text,
    price double,
    quantity int,
    PRIMARY KEY (id, locationId)
);

    environment text,
    event_log text,
    function text,
    identity text,
    level text,
    partner_transaction_id text,
    process_time float,
    product_id text,
    request text,
    request_id text,
    response_id text,
    response_size int,
    serial_number text,
    server_name text,
    server_version text,
    service text,
    site_code text,
    source text,
    status text,
    status_code int,
    status_message text,
    tsd_timestamp bigint,
    type text,
    url text,
    user_company text,
    user_name text,
    PRIMARY KEY (id, event_time)
